                    function numberOfPairs(gloves) {
                      let res = 0;
                      let obj = {};

                      for (let i = 0; i < gloves.length; i++) {
                        let current = gloves[i];
                        if (!obj[current]) {
                          obj[current] = 1;
                        } else obj[current]++;
                      }

                      for (let prop in obj) {
                        obj[prop] = Math.floor(obj[prop] / 2);
                        res += obj[prop];
                      }

                      return res;
                    }
